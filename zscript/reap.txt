class Reapose : Actor
{
    Default
    {
        Radius 30;
        Height 56;
        Mass 400;
        Speed 8;
        Health 200;
        PainChance 128;
        MONSTER;
        +FLOAT;
        +NOGRAVITY;
        +NOICEDEATH;

		Scale 0.1;

        SeeSound "cacodemon/sight";
        AttackSound "cacodemon/attack";
        PainSound "cacodemon/pain";
        DeathSound "cacodemon/death";
    }

    States
    {
    Spawn:
        REAP A 4 Bright A_Look;
        REAP B 4 Bright A_Look;
        REAP C 4 Bright A_Look;
        REAP D 4 Bright A_Look;
        Loop;

    See:
        REAP A 4 Bright A_Chase;
        REAP B 4 Bright A_Chase;
        REAP C 4 Bright A_Chase;
        REAP D 4 Bright A_Chase;
        Loop;

	Missile:
		REAP D 2 Bright;
		REAP E 2 Bright;
		REAP F 2 Bright;
		REAP G 2 Bright;
		REAP H 6 Bright A_FaceTarget;
		TNT1 A 0 A_PlaySound("reapose/shoot", CHAN_WEAPON);
		TNT1 A 0 A_CustomMissile("CacodemonBall", 32, 0, 0);
		REAP G 2 Bright;
		REAP F 2 Bright;
		REAP E 2 Bright;
		REAP D 2 Bright;
		Goto See;

	Death:
		REAP E 4 A_Scream;
		REAP F 4;
		REAP G 4;
		REAP H 4;
		REAP G 4 A_NoBlocking;
		REAP F 4;
		REAP E 4;
		REAP D 6;
		REAP C 6;
		REAP B 8;
		REAP A 10;
		TNT1 A 0;
		TNT1 A 0 StartShrinkThinker;


		REAP H -1;
		Stop;




    }
	action void StartShrinkThinker()
	{
		ScaleThinker.Create(self, Scale.X, Scale.Y, 0.01, 0.005, 70);
	}

}
